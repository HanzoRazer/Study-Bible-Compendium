Put this file in your repo root next to:

#!/usr/bin/env python3
"""
Import milestone-scoped JSON issue files into GitHub:
- Creates missing labels
- Creates/fetches milestone "Production Readiness v1.1"
- Creates issues from JSON (title/body/labels)
- Assigns each created issue to milestone
- Captures created issue numbers + URLs into created_issues_v1_1.csv

Requirements:
- GitHub CLI: gh
- Auth: gh auth login
- Python 3.8+
"""

from __future__ import annotations

import csv
import json
import os
import subprocess
import sys
from dataclasses import dataclass
from pathlib import Path
from typing import Any, Dict, List, Optional, Tuple


MILESTONE_TITLE = "Production Readiness v1.1"
DEFAULT_LABEL_COLOR = "ededed"  # light gray
OUTPUT_CSV = "created_issues_v1_1.csv"

JSON_FILES = [
    "milestone_v1_1_fts.json",
    "milestone_v1_1_pagination.json",
    "milestone_v1_1_performance.json",
]


@dataclass
class CreatedIssue:
    number: int
    url: str
    title: str
    source_file: str


def run(cmd: List[str], *, input_text: Optional[str] = None) -> str:
    """Run a command, return stdout, raise on error."""
    proc = subprocess.run(
        cmd,
        input=input_text.encode("utf-8") if input_text is not None else None,
        stdout=subprocess.PIPE,
        stderr=subprocess.PIPE,
        check=False,
    )
    if proc.returncode != 0:
        raise RuntimeError(
            f"Command failed ({proc.returncode}): {' '.join(cmd)}\n"
            f"STDOUT:\n{proc.stdout.decode('utf-8', errors='replace')}\n"
            f"STDERR:\n{proc.stderr.decode('utf-8', errors='replace')}\n"
        )
    return proc.stdout.decode("utf-8", errors="replace")


def require_gh_auth() -> None:
    try:
        out = run(["gh", "auth", "status"])
        if "Logged in to github.com" not in out:
            raise RuntimeError("Not authenticated with gh.")
    except Exception as e:
        raise RuntimeError(
            "GitHub CLI not authenticated. Run:\n\n  gh auth login\n"
        ) from e


def parse_repo(repo: Optional[str]) -> str:
    """Return OWNER/REPO."""
    if repo:
        return repo
    # Try to infer from gh
    out = run(["gh", "repo", "view", "--json", "nameWithOwner", "-q", ".nameWithOwner"])
    return out.strip()


def gh_api_json(args: List[str], *, input_obj: Optional[Dict[str, Any]] = None) -> Any:
    """
    Call gh api and parse JSON response.
    args example: ["repos/OWNER/REPO/labels", "--method", "GET"]
    """
    cmd = ["gh", "api"] + args
    if input_obj is not None:
        cmd += ["--input", "-"]
        raw = run(cmd, input_text=json.dumps(input_obj))
    else:
        raw = run(cmd)
    raw = raw.strip()
    if not raw:
        return None
    try:
        return json.loads(raw)
    except json.JSONDecodeError:
        # Some gh outputs are not JSON if -q used incorrectly
        return raw


def ensure_labels(repo: str, label_names: List[str]) -> None:
    # Get existing labels
    existing = gh_api_json([f"repos/{repo}/labels?per_page=100", "--method", "GET"])
    existing_names = {lbl["name"] for lbl in (existing or [])}

    for name in sorted(set(label_names)):
        if name in existing_names:
            continue
        print(f"[label] creating: {name}")
        gh_api_json(
            [f"repos/{repo}/labels", "--method", "POST"],
            input_obj={"name": name, "color": DEFAULT_LABEL_COLOR},
        )


def get_or_create_milestone(repo: str, title: str) -> int:
    # list open milestones
    milestones = gh_api_json([f"repos/{repo}/milestones?state=open&per_page=100", "--method", "GET"]) or []
    for ms in milestones:
        if ms.get("title") == title:
            return int(ms["number"])

    # create milestone
    print(f"[milestone] creating: {title}")
    created = gh_api_json(
        [f"repos/{repo}/milestones", "--method", "POST"],
        input_obj={"title": title},
    )
    return int(created["number"])


def load_issues_from_file(path: Path) -> List[Dict[str, Any]]:
    data = json.loads(path.read_text(encoding="utf-8"))
    if not isinstance(data, list):
        raise ValueError(f"{path} is not a JSON array.")
    for i, item in enumerate(data):
        if "title" not in item or "body" not in item:
            raise ValueError(f"{path}: item {i} missing title/body.")
        if "labels" in item and not isinstance(item["labels"], list):
            raise ValueError(f"{path}: item {i} labels must be a list.")
    return data


def create_issue(repo: str, title: str, body: str, labels: List[str], milestone_number: int) -> Tuple[int, str]:
    created = gh_api_json(
        [f"repos/{repo}/issues", "--method", "POST"],
        input_obj={
            "title": title,
            "body": body,
            "labels": labels,
            "milestone": milestone_number,
        },
    )
    return int(created["number"]), str(created["html_url"])


def main() -> int:
    import argparse

    ap = argparse.ArgumentParser(description="Import v1.1 milestone issue JSON files into GitHub.")
    ap.add_argument("--repo", help="OWNER/REPO. If omitted, inferred from current repo.", default=None)
    ap.add_argument("--dry-run", action="store_true", help="Validate + show plan, do not create anything.")
    ap.add_argument("--files", nargs="*", default=JSON_FILES, help="JSON files to import (default: v1.1 splits).")
    args = ap.parse_args()

    require_gh_auth()
    repo = parse_repo(args.repo)

    # Load all issues
    all_issues: List[Tuple[str, Dict[str, Any]]] = []
    all_labels: List[str] = []
    for f in args.files:
        p = Path(f)
        if not p.exists():
            raise FileNotFoundError(f"Missing file: {f}")
        issues = load_issues_from_file(p)
        for item in issues:
            all_issues.append((f, item))
            all_labels.extend(item.get("labels", []))

    print(f"[plan] repo={repo}")
    print(f"[plan] milestone='{MILESTONE_TITLE}'")
    print(f"[plan] files={args.files}")
    print(f"[plan] issues={len(all_issues)}")
    print(f"[plan] unique labels={len(set(all_labels))}")

    if args.dry_run:
        print("[dry-run] No changes made.")
        return 0

    # Ensure labels + milestone
    ensure_labels(repo, list(set(all_labels)))
    milestone_number = get_or_create_milestone(repo, MILESTONE_TITLE)
    print(f"[milestone] using number: {milestone_number}")

    created: List[CreatedIssue] = []
    for source_file, item in all_issues:
        title = item["title"].strip()
        body = item["body"]
        labels = item.get("labels", [])
        print(f"[create] {title} ({source_file})")
        number, url = create_issue(repo, title, body, labels, milestone_number)
        created.append(CreatedIssue(number=number, url=url, title=title, source_file=source_file))
        print(f"  -> #{number} {url}")

    # Write CSV log
    with open(OUTPUT_CSV, "w", newline="", encoding="utf-8") as fp:
        w = csv.writer(fp)
        w.writerow(["issue_number", "url", "title", "source_file"])
        for ci in created:
            w.writerow([ci.number, ci.url, ci.title, ci.source_file])

    print(f"[done] created {len(created)} issues")
    print(f"[done] log written: {OUTPUT_CSV}")
    return 0


if __name__ == "__main__":
    try:
        raise SystemExit(main())
    except Exception as e:
        print(f"[error] {e}", file=sys.stderr)
        raise
